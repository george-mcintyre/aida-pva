267 * Convert an arbitrary value to a {@link PVStructure} containing an `NTScalar`
268 * <p>
269 * The value must be the java equivalent of one of the supported AIDA-PVA scalar types
270 * - {@link Boolean}, {@link Byte}, {@link Short},
271 * - {@link Integer}, {@link Long}, {@link Float},
272 * - {@link Double} or {@link String}
273 *
274 * @param value java equivalent of one of the supported AIDA-PVA scalar types
275 * @return {@link PVStructure} containing an NTScalar with the value
276 */
277 public static PVStructure asScalar(Object value) {
278 // Null values are returned as an empty {@link PVStructure}
279 if (value == null) {
280 return NT_SCALAR_EMPTY_STRUCTURE;
281 }
282 // Find out the AIDA-PVA type that corresponds to the given value
283 AidaType aidaType = aidaTypeOf(value);
284 if (aidaType == null) {
285 throw new AidaInternalException("Could not determine type of returned value: " + value);
286 }
287 // And find out the corresponding PVScalar type that it needs to converted into
288 ScalarType scalarType = scalarTypeOf(aidaType);
289 // Create the field of that type to be added to the {@link PVStructure} we return
290 Field[] scalarField = new Field[] {FieldFactory.getFieldCreate().createScalar(scalarType)};
291 // Create a new {@link PVStructure} to return containing the field we created
292 PVStructure retVal = PVFactory.getPVDataCreate().createPVStructure(
293 // Structure with one element: label and empty scalar field definition
294 FieldFactory.getFieldCreate().createStructure(NTSCALAR_ID, NT_SCALAR_TOP_STRUCTURE_NAME, scalarField));
295 // In the second phase of {@link PVStructure} creation we now set the field value
296 // Note that the field name to set is always called "value" for NT types
297 setValue(retVal, value, aidaType);
298 return retVal;

336 /**
337 * Convert an arbitrary list of homogeneously sized columns of values to a {@link PVStructure} containing an `NTTable`
338 * The value of each of the columns must be a java {@link List} of a type equivalent of one of the supported AIDA-PVA scalar types
339 * - {@link Boolean}, {@link Byte}, {@link Short},
340 * - {@link Integer}, {@link Long}, {@link Float},
341 * - {@link Double} or {@link String}
342 *
343 * @param values the list of values
344 * @param aidaChannelConfig the configuration
345 * @return the returned PVStructure containing the NT_TABLE
346 */
347 public static PVStructure asNtTable(List<List<Object>> values, AidaChannelConfig aidaChannelConfig) {
348 // If there is nothing to add or that the list is empty or if the columns are empty return an empty
349 // {@link PVStructure}
350 if (values == null || values.isEmpty() || values.get(0).isEmpty()) {
351 return NT_TABLE_EMPTY_STRUCTURE;
352 }
353 // If the lists are not homogeneously sized then raise an exception
354 int columnSize = values.get(0).size();
355 for (List<Object> column : values) {
356 if (column.size() != columnSize) {
357 throw new AidaInternalException("Columns in NTTable not homogeneously sized. Normal size: " + columnSize + ", One column size: " + column.size());
358 }
359 }
360 // The field names, labels and types for an NTTable come from the Channel Configuration specified in the
361 // CHANNELS.YML file loaded when the service initialises.
362 // We need to retrieve these configured values from the given configuration associated with the
363 // request we're processing
364 List<String> fieldNames = new ArrayList<String>();
365 List<String> labels = new ArrayList<String>();
366 List<AidaType> aidaTypes = new ArrayList<AidaType>();
367 // We need to create a set of fields that we will fill with these values
368 List<Field> pvFields = new ArrayList<Field>();
369 setFieldsWithNamesLabelsAndTypesFromConfig(pvFields, aidaChannelConfig, values, fieldNames, labels, aidaTypes);
370 /// FIRST STAGE {@link PVStructure} creation: Structure
371 // List of strings that will contain the labels names
372 Field labelsArray = FieldFactory.getFieldCreate().createScalarArray(pvString);
373 // Create the second level {@link Field} that will conta
374 // top fields, this one holding the structure of the rest of the NTTable
375 // - fieldNames = PVScalarArray containing pvstring and
376 // - columns of values = A set of Fields that are each PVScalarArrays
377 Field structure = FieldFactory.getFieldCreate().createStructure(toStringArray(fieldNames), toFieldArray(pvFields));
378 // Create the two top fields of the
379 Field[] topFields = new Field[] {labelsArray, structure};
380 // Create the top {@link PVStructure} that will be returned (labels and values)
381 PVStructure retVal = PVFactory.getPVDataCreate().createPVStructure(FieldFactory.getFieldCreate().createStructure(NTTABLE_ID, NT_TABLE_TOP_STRUCTURE_NAMES, topFields));
382 /// SECOND STAGE {@link PVStructure} creation: Set values
383 // Set the field labels
384 setValues(retVal, NT_LABELS_NAME, labels, STRING_ARRAY);
385 // Set array values
386 for (int i = 0; i < fieldNames.size(); i++) {
387 String fieldName = NT_FIELD_NAME + "." + fieldNames.get(i);
388 List<?> columnValues = values.get(i);
389 AidaType aidaType = aidaTypes.get(i);
390 setValues(retVal, fieldName, columnValues, aidaType);
391 }
392 return retVal;
393 }

535 /**
536 * Extract the String value from the field
537 * Any time a double or float is found, add it to the appropriate map under the `fieldPath` key
538 *
539 * @param field the field
540 * @param fieldPath path under which this field sits. Used only for floats and doubles that need
541 * to set the path so the name in the pair is prefixed with this value if set
542 * @param floatArgumentList to return the list of floats found and the paths
543 * @param doubleArgumentList to return the list of doubles found and the paths
544 * @return the extracted value
545 * @throws RPCRequestException of the field type is unsupported
546 */
547 public static String fieldToString(PVField field, String fieldPath, List<FloatArgument> floatArgumentList, List<DoubleArgument> doubleArgumentList) throws RPCRequestException {
548 String value;
549 if (field instanceof PVBoolean) {
550 value = Boolean.toString(((PVBoolean) field).get());
551 } else if (field instanceof PVByte) {
552 value = Byte.toString(((PVByte) field).get());
553 } else if (field instanceof PVUByte) {
554 value = Byte.toString(((PVUByte) field).get());
555 } else if (field instanceof PVShort) {
556 value = Short.toString(((PVShort) field).get());
557 } else if (field instanceof PVInt) {
558 value = Integer.toString(((PVInt) field).get());
559 } else if (field instanceof PVLong) {
560 value = Long.toString(((PVLong) field).get());
561 } else if (field instanceof PVFloat) {
562 // For float store the ieee value in the list as well
563 value = floatFieldToString((PVFloat) field, fieldPath, floatArgumentList);
564 } else if (field instanceof PVDouble) {
565 // For double store the ieee value in the list as well
566 value = doubleFieldToString((PVDouble) field, fieldPath, doubleArgumentList);
567 } else if (field instanceof PVString) {
568 // For strings quote them is they are subfields but leave them untouched if not
569 value = stringFieldToString((PVString) field, fieldPath);
570 } else if (field instanceof PVStructure) {
571 // For structures recurse for each of the fields
572 value = structureFieldToString((PVStructure) field, fieldPath, floatArgumentList, doubleArgumentList);
573 } else if (field instanceof PVArray) {
574 // For arrays recurse for each of the array elements
575 value = arrayFieldToString((PVArray) field, fieldPath, floatArgumentList, doubleArgumentList);
576 } else {
577 throw new RPCRequestException(ERROR, "Invalid argument value: can only accept scalar, scalar array, structure or structure array");
578 }
579 return value;
580 }

593 private static String arrayFieldToString(PVArray field, String fieldPath, List<FloatArgument> floatArgumentList, List<DoubleArgument> doubleArgumentList) throws RPCRequestException {
594 String value;
595 StringBuilder arrayBuilder = new StringBuilder();
596 arrayBuilder.append("["); // Start of array
597 boolean firstTime = true;
598 // Convert all array entries to Strings, recurse to get any floating point numbers into the appropriate lists
599 List<String> arrayList = arrayFieldToStringList(field, fieldPath, floatArgumentList, doubleArgumentList);
600 for (String subValue : arrayList) {
601 // For each string value in the retrieved list
602 // Comma separate entries
603 if (!firstTime) {
604 arrayBuilder.append(", ");
605 }
606 firstTime = false;
607 // Add to list
608 arrayBuilder.append(subValue);
609 }
610 arrayBuilder.append("]"); // End of array
611 value = arrayBuilder.toString();
612 return value;

615 /**
616 * Convert the {@link PVStructure} field to the json string representation of a structure of that field,
617 * storing any ieee values found unchanged in the given {@link FloatArgument}and {@link DoubleArgument} lists
618 *
619 * @param field the {@link PVStructure} field
620 * @param fieldPath the absolute path to the field from the root
621 * @param floatArgumentList the {@link FloatArgument} list to store any ieee values
622 * @param doubleArgumentList the {@link DoubleArgument} list to store any ieee values
623 * @return the json string representation of the {@link PVStructure}
624 * @throws RPCRequestException if anything bad happens
625 */
626 public static String structureFieldToString(PVStructure field, String fieldPath, List<FloatArgument> floatArgumentList, List<DoubleArgument> doubleArgumentList) throws RPCRequestException {
627 String value;
628 StringBuilder structureBuilder = new StringBuilder();
629 structureBuilder.append("{"); // Start of structure
630 boolean firstTime = true;
631 for (PVField subField : field.getPVFields()) {
632 // For each field in structure
633 String elementName = subField.getFieldName(); // get the field name
634 String absolutePathToElement = fieldPath + "." + elementName; // This is the absolute path to the element from the root of the structure
635 // And recurse into the field to get the string representation of it - of course any floats it contains will be stored in the appropriate lists
636 String elementValue = fieldToString(subField, absolutePathToElement, floatArgumentList, doubleArgumentList);
637 addStructureElementValue(structureBuilder, elementName, elementValue, firstTime);
638 firstTime = false;
639 }
640 structureBuilder.append("}"); // End of structure
641 value = structureBuilder.toString();
642 return value;
643 }

651 * @param firstElement is this the first element added to the structure

654 // Comma separate entries in the structure
655 if (!firstElement) {
656 structureBuilder.append(", ");
657 }
658 // Add the element name and the element value
659 structureBuilder.append("\"");
660 structureBuilder.append(elementName);

676 List<String> arrayList = new ArrayList<String>();

691 longArrayToString((PVLongArray) array, arrayList);
692 } else if (array instanceof PVFloatArray) {
