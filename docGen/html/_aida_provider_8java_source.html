74 * <p>
75 * First, make sure that all config is loaded.
76 * <p>
77 * Channels are matched literally but also by using the {@link WildcardMatcher#match}
78 * to see if any of the patterns specified as supported by this channel correspond to the

81 * @param channelName the channel name from the request
82 * @return the {@link AidaChannel} object that matches the given channel name
83 */
84 public AidaChannel getAidaChannel(String channelName) {
85 loadChannelMapIfNotLoaded();
86 // Look for an exact match or a {@link WildcardMatcher#match} if that fails
87 AidaChannel aidaChannel = this.channelMap.get(channelName);
88 if (aidaChannel == null) {
89 for (Map.Entry<String, AidaChannel> entry : this.channelMap.entrySet()) {
90 if (WildcardMatcher.match(entry.getKey(), channelName)) {
91 aidaChannel = entry.getValue();
92 break;
93 }
94 }
95 }
96 return aidaChannel;

99 /**
100 * For speed the list of channels and their configurations are cached in a memory resident
101 * {@link HashMap}. This method is used to prime that {@link HashMap} from the list of
102 * {@link AidaProvider#configurations} that have been loaded in from the Channel Provider's
103 * CHANNELS.YML file.
104 */
105 private void loadChannelMapIfNotLoaded() {
106 /// Make sure that no other thread does this at the same time
107 synchronized (this.channelMap) {
108 // Only load the map if it is empty
109 if (this.channelMap.isEmpty()) {
110 // Get all the configuration groups
111 for (AidaConfigGroup configuration : getConfigurations()) {
112 AidaChannelConfig getterConfig = configuration.getGetterConfig();
113 AidaChannelConfig setterConfig = configuration.getSetterConfig();
114 // In the getter config, set the labels to be the field names, if the labels are not specified
115 if (getterConfig != null) {
116 setDefaultLabels(getterConfig.getFields());
117 }
118 // In the setter config, set the labels to be the field names, if the labels are not specified
119 if (setterConfig != null) {
120 setDefaultLabels(setterConfig.getFields());
121 }
122 // Set give all channels that are in this configuration group the same config
123 for (String channelName : configuration.getChannels()) {
124 AidaChannel aidaChannel = new AidaChannel(channelName, configuration.getGetterConfig(), configuration.getSetterConfig());
125 this.channelMap.put(channelName, aidaChannel);
126 // Get index of last separator using new and legacy format channel names
127 int indexOfLastSeparator = channelName.lastIndexOf(":");
128 int indexOfLastLegacySeparator = channelName.lastIndexOf("//");
129 // Add legacy style channel name as well as new style, regardless as to how it is specified in the channels file
130 // If specified using legacy separator then add entry with the new style as well
131 if (indexOfLastLegacySeparator != -1) {
132 this.channelMap.put(channelName.substring(0, indexOfLastLegacySeparator) + ":" + channelName.substring(indexOfLastLegacySeparator + 2), aidaChannel);
133 } else
134 // If specified with new naming style then add an entry with the legacy separator for backwards compatibility
135 if (indexOfLastSeparator != -1) {
136 String legacyStyle = null;
137 String[] parts = channelName.split(":");
138 int nParts = parts.length;
139 if (nParts == 3) {
140 // If uri has three parts like for MAGNET then need to add legacy separator after second not first
141 legacyStyle = parts[0] + "//" + parts[1] + ":" + parts[2];
142 } else {
143 // Otherwise, add before last part
144 legacyStyle = channelName.substring(0, indexOfLastSeparator) + "//" + channelName.substring(indexOfLastSeparator + 1);
145 }
146 this.channelMap.put(legacyStyle, aidaChannel);
147 }
148 }
149 }
150 }
