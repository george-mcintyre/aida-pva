29/**
30 * Implementation of a Channel Provider.
31 * A Channel Provider is responsible for communicating with the Native Channel Provider
32 * to get and receive information from the AIDA-PVA network.
33 * <p>
34 * This class encodes generic functionality common to all Channel Providers on the AIDA-PVA network.
35 * Specific customisations should be coded in the {@link edu.stanford.slac.aida.impl.AidaChannelProvider}
36 * class.
37 * <p>
38 * This class automatically calls the initialisation of the underlying Native Provider
39 * and shows the AIDA-PVA banner.
40 * <p>
41 * It also handles the five main classes of requests routing them to the correct underlying
42 * - {@link slac.aida.NativeChannelProvider} entry point:
43 * - {@link ChannelProvider#requestScalar(String, AidaArguments, AidaType)}
44 * - {@link ChannelProvider#requestScalarArray(String, AidaArguments, AidaType)}
45 * - {@link ChannelProvider#requestTable(String, AidaArguments)}
46 * - {@link ChannelProvider#setValue(String, AidaArguments)}
47 * - {@link ChannelProvider#setValueWithResponse(String, AidaArguments)}
48 */
49public abstract class ChannelProvider extends NativeChannelProvider {
50 /**
51 * Logger to log info
52 */
53 private static final Logger logger = Logger.getLogger(ChannelProvider.class.getName());
54
55 private final AidaProvider aidaProvider;
56
57 /**
58 * Constructor for the channel provider that reads the configuration from environment variables and
59 * configuration files.
60 * <p>
61 * Uses {@link ChannelProviderFactory#create(ChannelProvider)} to check
62 * the `aida.pva.channels.filename` property or the `AIDA_PVA_CHANNELS_FILENAME`
63 * environment variable to load the yaml configuration file
64 * <p>
65 * Lists the abbreviated list of hosted channels to the console.
66 * <p>
67 * Calls the {@link ChannelProvider#logHostedChannels()}
68 */
69 public ChannelProvider() {
70 // Aida server Banner on standard output
71 System.out.println(AidaPva.banner());
72
73 // Create the channel based on environment and configuration files
74 this.aidaProvider = ChannelProviderFactory.create(this);
75
76 // Log the configuration of the channel we just created
77 logHostedChannels();
78
79 // Call initialisation for the channel provider
80 aidaServiceInit();
81 }

83 /**
84 * Handles scalar requests by calling the appropriate Native Method. This can return :-
85 * - scalars e.g. Boolean, Integer, Byte, Long, Float, Double, ...
86 *
87 * @param channelName request channel name
88 * @param arguments request arguments
89 * @param aidaType the return type
90 * @return object
91 */
92 public Object requestScalar(String channelName, AidaArguments arguments, AidaType aidaType) {
93 switch (aidaType) {
94 case BOOLEAN:
95 return aidaRequestBoolean(channelName, arguments);
96 case BYTE:
97 return aidaRequestByte(channelName, arguments);
98 case SHORT:
99 return aidaRequestShort(channelName, arguments);
100 case INTEGER:
101 return aidaRequestInteger(channelName, arguments);
102 case LONG:
103 return aidaRequestLong(channelName, arguments);
104 case FLOAT:
105 return aidaRequestFloat(channelName, arguments);
106 case DOUBLE:
107 return aidaRequestDouble(channelName, arguments);
108 case STRING:
109 return aidaRequestString(channelName, arguments);
110 }
111 return null;
112 }
113
114 /**
115 * Handles scalar array requests by calling the appropriate Native Method. This can return :-
116 * - collections of scalars e.g. List of {@link Boolean}, List of {@link Integer} , List of {@link Byte} , List of {@link Long} , List of {@link Float}, List of {@link Double}, ...
117 *
118 * @param channelName request channel name
119 * @param arguments request arguments
120 * @param aidaType the scalar type underpinning this scalar array
121 * @return List of scalar object
122 */
123 public List<?> requestScalarArray(String channelName, AidaArguments arguments, AidaType aidaType) {
124 switch (aidaType) {
125 case BOOLEAN_ARRAY:
126 List<Boolean> tList = new ArrayList<Boolean>();
127 for (Boolean b : aidaRequestBooleanArray(channelName, arguments)) {
128 tList.add(b);
129 }
130 return tList;
131 case BYTE_ARRAY:
132 List<Byte> bList = new ArrayList<Byte>();
133 for (Byte b : aidaRequestByteArray(channelName, arguments)) {
134 bList.add(b);
135 }
136 return bList;
137 case SHORT_ARRAY:
138 List<Short> sList = new ArrayList<Short>();
139 for (Short s : aidaRequestShortArray(channelName, arguments)) {
140 sList.add(s);
141 }
142 return sList;
143 case INTEGER_ARRAY:
144 List<Integer> iList = new ArrayList<Integer>();
145 for (Integer i : aidaRequestIntegerArray(channelName, arguments)) {
146 iList.add(i);
147 }
148 return iList;
149 case LONG_ARRAY:
150 List<Long> lList = new ArrayList<Long>();
151 for (Long l : aidaRequestLongArray(channelName, arguments)) {
152 lList.add(l);
153 }
154 return lList;
155 case FLOAT_ARRAY:
156 List<Float> fList = new ArrayList<Float>();
157 for (Float f : aidaRequestFloatArray(channelName, arguments)) {
158 fList.add(f);
159 }
160 return fList;
161 case DOUBLE_ARRAY:
162 List<Double> dList = new ArrayList<Double>();
163 for (Double d : aidaRequestDoubleArray(channelName, arguments)) {
164 dList.add(d);
165 }
166 return dList;
167 case STRING_ARRAY:
168 return Arrays.asList(aidaRequestStringArray(channelName, arguments));
169 }
170 return null;
171 }
