33 public static AidaProvider create(ChannelProvider channelProvider) {
34 // Get service name and channel definitions for the server to publish.
35 // Priority: max=properties, medium=environment, low=default
36 String channelsFilename = System.getProperty("aida.pva.channels.filename", CHANNELS_FILENAME_DEFAULT);
37 String channelsFilenameFromEnv = System.getenv("AIDA_PVA_CHANNELS_FILENAME");
38 if (channelsFilenameFromEnv != null) {
39 channelsFilename = channelsFilenameFromEnv;
40 }
41 if ( !channelsFilename.equals(CHANNELS_FILENAME_DEFAULT) ) {
42 logger.info("Loading channel configuration from: " + channelsFilename);
43 }
44
45 // Set up the object mapper to read the channels
46 ObjectMapper mapper = new ObjectMapper(new YAMLFactory());
47 try {
48 File channelSource = new File(channelsFilename);
49 AidaProvider aidaProvider;
50 if (channelSource.exists()) {
51 aidaProvider = mapper.readValue(channelSource, AidaProvider.class);
52 } else {
53 throw new RuntimeException("Can't access channel provider configuration file: " + channelsFilename);
54 }
55 aidaProvider.setChannelProvider(channelProvider);
56 return aidaProvider;
57 } catch (Exception e) {
58 logger.severe("Unable to initialise channel provider with " + channelsFilename + " : " + e.getMessage());
59 throw new RuntimeException(e.getMessage());
60 }
61 }
