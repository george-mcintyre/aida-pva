146/**
147 * An Arguments structure stores all of the arguments passed from the request to the Native Channel Provider.
148 * It contains a count of the total number of arguments - `argumentCount` - and a pointer
149 * to that many Arguments.
150 * Arguments from the client are stored as strings, and converted to whatever type is required whenever
151 * they are used. But floating points are different because they require more precision and significance
152 * than string representations can give. For this reason floating points are additionally passed in the
153 * FloatingPointValue array where `floatingPointValuesCountCount` is the count of how many of them there are.

163/**
164 * Table structure.
165 * This structure holds everything that a Native Channel Provider needs for returning a Table
166 * to the client. A table is a set of homogeneously sized vectors representing columns of data.
167 * The configuration of the table (labels, names, descriptions, etc) is handled outside of the
168 * Native Channel Provider, so all the implementation needs to do is provide the raw data for the
169 * columns.
170 *
171 * Implementors don't manipulate these structures directly as AIDA-PVA Module provides
172 * table manipulation functions:
173 * - tableCreate.
174 * - tableAddColumn.
175 * - tableAddSingleRowFloatColumn.
176 * - tableAddSingleRowLongColumn.
177 * - tableAddSingleRowBooleanColumn.
178 * - tableAddSingleRowByteColumn.
179 * - tableAddSingleRowShortColumn.
180 * - tableAddSingleRowIntegerColumn.
181 * - tableAddSingleRowDoubleColumn.
182 * - tableAddSingleRowStringColumn.
183 * - tableAddFixedWidthStringColumn.
184 * - tableAddStringColumn.
185 */
186typedef struct
187{
188 int columnCount; ///< number of columns in table
189 int rowCount; ///< number of rows in table
190 Type* types; ///< the scalar type of each column, one of BOOLEAN, BYTE, SHORT, INTEGER, LONG, FLOAT, DOUBLE, or STRING
191 void** ppData; ///< the data. an array of [rows][columns]
192 int _currentColumn; ///< For internal use by addColumn() etc
193} Table;

278/**
279 * ascanf, avscanf
280 *
281 * Synopsis
282 *
283 * int ascanf(Arguments *arguments, const char *format, ...);
284 * int avscanf(Arguments *arguments, Value *value, const char *format, ...);
285 *
286 * Details
287 * Reads data from the given arguments and stores them according to parameter format into the locations given by the additional arguments, as if scanf was used, but reading from arguments instead of the standard input (stdin).
288 *
289 * The additional arguments should point to already allocated objects of the type specified by their corresponding format specifier. For strings and arrays only the pointer needs to be pre-allocated.
290 *
291 * The only space allocated by this function is for strings, or arrays. You need to free those when you're done
292 *
293 * Note, only the provided pointer needs to be freed as only one allocation is made e.g.
294 *
295 * Arguments arguments;
296 * float theFloat;
297 * int *intArray;
298 * ascanf(arguments "$of, %da", "f", &theFloat, "fooArray", &intArray);
299 * ///< Do stuff
300 *
301 * Differences from scanf
302 * There are a number of other differences from scanf which are best described by example:
303 *
304 * 1. Scan into simple variable
305 *
306 * int n;
307 * ascanf("%d", "NPOS", &n);
308 *
309 * You must always provide the name of the variable and the pointer to the place to put the value in pairs
310 *
311 * 2. Optional arguments
312 * Optional arguments are specified with the o character before the format character.
313 *
314 * short flag = 10; ///< 10 is the default value
315 * ascanf("%ohd", "flag", &flag);
316 *
317 * By default all arguments are considered required unless this character is specified. For optional arguments the pointer provided must point to the default value. In the case of arrays and strings this will be copied into allocated storage that will need to be freed as normal. i.e. strings themselves don't need to be freed.
318 *
319 * Variable Names
320 * 1. You can specify simple variable names
321 * int simpleInt;
322 * ascanf(&arguments "%d, "simple", &simpleInt);
323 *
324 * 1. You can specify simple names or you can use dot and square brace notation to refer to arguments that refer to json structures. e.g. given a variable named json and presented as
325 *
326 * json=' { "foo": {"bar": 0} }}'
327 *
328 * You can specify the name as json.foo.bar to retrieve the 0 value *
329 *
330 * 2. Also given a variable named jsonArray and presented as
331 * jsonArray=' [ {"foo": 10}, {"bar": 20} ]'
332 *
333 * You can specify the name as jsonArray[1].bar to retrieve the 20 value
334 *
335 * 3. Finally if you use the name value in the the avscanf() function will use the supplied value to get the data for that parameter
336 * Arguments arguments;
337 * Value value;
338 * int *intArray;
339 * avscanf(&arguments &value, "%da, "fooArray", &intArray);
340 * ///< Do stuff
341 * free(intArray);
342 *
343 * Format Specifiers
344 * Supported formats specifiers
345 * - d : int * - extract an integer into the corresponding variable (see l & h below).
346 * - u : unsigned int * - extract an unsigned integer into the corresponding variable (see l & h below)
347 * - f : float * - extract a floating point number (see l below)
348 * - s : char * - extract a string of characters into allocated space and point the corresponding variable to it
349 * - c : char * - extract a single character into the corresponding variable
350 *
351 * Required flag
352 * - o - optional precede the format with 'o' to indicate that the argument is optional
353 *
354 * Prefixes
355 * - h - short * - preceding d will retrieve a short e.g. %hd
356 * - l - long *, double * - preceding d will retrieve a long eg. %ld, preceding f will retrieve a double eg. %lf
357 *
358 * Suffixes
359 * - a - extract an array of the preceding type into a block of allocated space and point the corresponding variable to it. the variable will have an extra level of indirection than the non-array version e.g. "%d" "int *" becomes "%da" "int **"
360 *
361 * @param env
362 * @param arguments arguments that the function processes as its source to retrieve the data.
363 * @param value value that the function processes as its source to retrieve the data
364 * @param formatString C string that contains a format string as described above
365 * @param ... Depending on the format string, the function may expect a sequence of additional arguments,
366 * containing pairs of names and pointers to allocated storage (except as indicated above),
367 * where the interpretation of the extracted data is stored with the appropriate type.
368 * There should be at least as many pairs of these arguments as the number of values stored
369 * by the format specifiers.
370 * Additional arguments are ignored by the function
371 * @return true if all required arguments were read and no errors occurred
372 * @throw MissingRequiredArgument if one of the required arguments are missing
373*/
374int ascanf(JNIEnv* env, Arguments* arguments, const char* formatString, ...);
375int avscanf(JNIEnv* env, Arguments* arguments, Value* value, const char* formatString, ...);
